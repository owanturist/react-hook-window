```tsx live noInline
import { useWindowedList } from 'react-hook-window'
import { Container, Item } from './components'

const CONTAINER_HEIGHT = 400
const ITEM_HEIGHT = 50
const items = Array.from({ length: 10000 }).map((_, i) => ({
  id: `item-${i}`,
  title: `Item #${i}`
}))

const FixedSizeList = () => {
  const { setRef, indexes, startSpace, endSpace } = useWindowedList({
    containerSize: CONTAINER_HEIGHT,
    itemSize: ITEM_HEIGHT,
    itemCount: items.length
  })

  return (
    <Container ref={setRef} height={CONTAINER_HEIGHT} overflowY="auto">
      <div style={{ height: startSpace }} />
      {indexes.map(index => {
        const item = items[index]

        return (
          <Item
            key={item.id}
            height={ITEM_HEIGHT}
            bg={index % 2 === 0 ? 'cool-gray-50' : 'cool-gray-100'}
          >
            {item.title}
          </Item>
        )
      })}
      <div style={{ height: endSpace }} />
    </Container>
  )
}

render(<FixedSizeList />)
```

## Horizontal scroll

```tsx live noInline
import { useWindowedList } from 'react-hook-window'
import { Container, Item, useResizeObserver } from './components'

const ITEM_WIDTH = 220
const items = Array.from({ length: 10000 }).map((_, i) => ({
  id: `item-${i}`,
  title: `Item #${i}`
}))

const FixedSizeList = () => {
  const [containerWidth, setContainerWidth] = React.useState(0)
  const { setRef, indexes, startSpace, endSpace, container } = useWindowedList({
    containerSize: containerWidth,
    itemSize: ITEM_WIDTH,
    itemCount: items.length,
    layout: 'horizontal'
  })

  useResizeObserver(node => {
    setContainerWidth(node.clientWidth)
  }, container)

  return (
    <Container
      ref={setRef}
      width="100%"
      display="flex"
      flexDirection="row"
      overflowX="auto"
    >
      <div style={{ flex: `0 0 ${startSpace}px` }} />
      {indexes.map(index => {
        const item = items[index]

        return (
          <Item
            key={item.id}
            flex={`0 0 ${ITEM_WIDTH}px`}
            height={160}
            bg={index % 2 === 0 ? 'cool-gray-50' : 'cool-gray-100'}
          >
            {item.title}
          </Item>
        )
      })}
      <div style={{ flex: `0 0 ${endSpace}px` }} />
    </Container>
  )
}

render(<FixedSizeList />)
```
